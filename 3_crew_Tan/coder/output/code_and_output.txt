def calculate_pi_approximation(terms):
    """
    Calculate an approximation of π using the Leibniz formula:
    π/4 = 1 - 1/3 + 1/5 - 1/7 + 1/9 - ...
    
    Parameters:
    terms (int): Number of terms to use in the series
    
    Returns:
    float: Approximation of π
    """
    total = 0.0
    
    for i in range(terms):
        # Calculate the term: (-1)^i / (2i + 1)
        term = (-1) ** i / (2 * i + 1)
        total += term
    
    # Multiply by 4 to get π approximation
    pi_approx = total * 4
    
    return pi_approx

# Calculate using 10,000 terms
terms = 10000
pi_approximation = calculate_pi_approximation(terms)

print(f"Calculating π using the Leibniz formula with {terms:,} terms")
print(f"π approximation: {pi_approximation}")
print(f"Actual π value: 3.141592653589793")
